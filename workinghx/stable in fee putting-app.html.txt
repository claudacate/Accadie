<script>
// ═══════════════════════════════════════════════
// STATE

var state = {
  stimp: 10.0,
  lengthFeet: 10,
  sensorActive: false,
  currentSide: 0,
  currentSideDir: 'R',
  currentUD: 0,
  currentUDDir: 'flat',
  locked: null,
  previous: null,
  manualMode: false,
  manualSide: 1.0,
  manualSideDir: 'R',
  manualUD: 1.0,
  manualUDDir: 'flat',
  isReading: false,
};

// ═══════════════════════════════════════════════
// SCREEN MANAGEMENT
// Three states: 'permission' | 'rotate' | 'app'
// ═══════════════════════════════════════════════
var permissionGranted = false;

function showScreen(which) {
  document.getElementById('permission-screen').classList.toggle('hidden', which !== 'permission');
  document.getElementById('rotate-overlay').classList.toggle('hidden', which !== 'rotate');
  var appRoot = document.getElementById('app-root');
  if (which === 'app') {
    appRoot.classList.add('visible');
  } else {
    appRoot.classList.remove('visible');
  }
}

function isLandscape() {
  return window.innerWidth > window.innerHeight;
}

function updateOrientation() {
  if (!permissionGranted) return;
  if (isLandscape()) {
    showScreen('app');
  } else {
    showScreen('rotate');
  }
}

window.addEventListener('resize', updateOrientation);
window.addEventListener('orientationchange', function() {
  setTimeout(updateOrientation, 150); // short delay for iOS to finish rotating
});

// ═══════════════════════════════════════════════
// PERMISSION
// ═══════════════════════════════════════════════
async function requestPermission() {
  if (typeof DeviceOrientationEvent !== 'undefined' &&
      typeof DeviceOrientationEvent.requestPermission === 'function') {
    try {
      var res = await DeviceOrientationEvent.requestPermission();
      if (res === 'granted') {
        permissionGranted = true;
        startSensor();
        updateOrientation();
      } else {
        alert('Motion access denied.\n\nGo to Settings → Safari → Motion & Orientation Access → ON, then reload.');
      }
    } catch(e) {
      alert('Error: ' + e.message);
    }
  } else if (typeof DeviceOrientationEvent !== 'undefined') {
    // Android / non-iOS — no permission needed
    permissionGranted = true;
    startSensor();
    updateOrientation();
  } else {
    alert('Motion sensors not supported. Use Manual entry.');
    permissionGranted = true;
    updateOrientation();
  }
}

// ═══════════════════════════════════════════════
// SENSOR — LANDSCAPE AXIS MAPPING
// ═══════════════════════════════════════════════
// In landscape orientation (home button right, screen up):
//   gamma = side slope: positive = right side up = L→R break → aim LEFT
//   beta  = up/down:    positive = front up = uphill  (away from ball)
//
// In landscape (home button left):
//   gamma flips sign, beta flips sign
//
// We detect orientation via window.orientation or screen.orientation.angle
// Standard landscape-right = 90°, landscape-left = -90° (270°)

function startSensor() {
  state.sensorActive = true;
  window.addEventListener('deviceorientation', onOrientation);
}

function onOrientation(e) {
  if (!state.isReading) return;

  var beta  = e.beta  || 0;
  var gamma = e.gamma || 0;

  // Detect landscape orientation angle
  var angle = 0;
  if (window.screen && window.screen.orientation) {
    angle = window.screen.orientation.angle || 0;
  } else if (typeof window.orientation !== 'undefined') {
    angle = window.orientation;
  }

  var side, sideDir, ud, udDir;

  if (angle === 90 || angle === -270) {
    // Landscape RIGHT (home button on right)
    // beta  = side slope (left/right tilt)
    // gamma = up/down (top/bottom tilt toward hole)
    side    = Math.abs(beta);
    sideDir = beta <= 0 ? 'R' : 'L';
    ud      = Math.abs(gamma);
    udDir   = Math.abs(gamma) < 0.4 ? 'flat' : (gamma <= 0 ? 'uphill' : 'downhill');

  } else if (angle === -90 || angle === 270) {
    // Landscape LEFT (home button on left) — signs flip
    side    = Math.abs(beta);
    sideDir = beta >= 0 ? 'L' : 'R';
    ud      = Math.abs(gamma);
    udDir   = Math.abs(gamma) < 0.4 ? 'flat' : (gamma >= 0 ? 'uphill' : 'downhill');

  } else {
    // Fallback
    side    = Math.abs(gamma);
    sideDir = gamma >= 0 ? 'R' : 'L';
    ud      = Math.abs(beta);
    udDir   = Math.abs(gamma) < 0.4 ? 'flat' : (gamma > 0 ? 'uphill' : 'downhill');
  }

  state.currentSide    = parseFloat(side.toFixed(1));
  state.currentSideDir = sideDir;
  state.currentUD      = parseFloat(ud.toFixed(1));
  state.currentUDDir   = udDir;

  // Update live display
  document.getElementById('live-side').textContent = state.currentSide.toFixed(1) + '%';
  document.getElementById('live-side-dir').textContent = sideDir === 'R' ? 'Aim Right →' : '← Aim Left';
  document.getElementById('live-ud').textContent = state.currentUD.toFixed(1) + '%';
  document.getElementById('live-ud-dir').textContent =
    udDir === 'flat' ? '— Flat' : udDir === 'uphill' ? '⬆ Uphill' : '⬇ Downhill';
}

function startReading() {
  state.isReading = true;
  showCenterState('reading');
  setBtn('read-btn', false);
  setBtn('lock-btn', true);
  setBtn('new-btn', false);
  setBtn('reset-btn', false);
  setBtn('avg-btn', false);
}

function lockReading() {
  state.isReading = false;
  if (state.locked) state.previous = Object.assign({}, state.locked);
  state.locked = {
    side:    state.currentSide,
    sideDir: state.currentSideDir,
    ud:      state.currentUD,
    udDir:   state.currentUDDir,
  };
  updateLockedDisplay();
  showCenterState('locked');
  setBtn('read-btn', false);
  setBtn('lock-btn', false);
  setBtn('new-btn', true);
  setBtn('reset-btn', true);
  setBtn('avg-btn', !!state.previous);
  recalc();
}

function averageReadings() {
  if (!state.locked || !state.previous) return;
  state.locked = {
    side:    parseFloat(((state.locked.side + state.previous.side) / 2).toFixed(1)),
    sideDir: state.locked.sideDir,
    ud:      parseFloat(((state.locked.ud + state.previous.ud) / 2).toFixed(1)),
    udDir:   state.locked.udDir,
  };
  state.previous = null;
  updateLockedDisplay();
  setBtn('avg-btn', false);
  recalc();
}

function resetReader() {
  state.locked = null;
  state.previous = null;
  state.isReading = false;
  showCenterState('ready');
  setBtn('read-btn', true);
  setBtn('lock-btn', false);
  setBtn('new-btn', false);
  setBtn('reset-btn', false);
  setBtn('avg-btn', false);
  clearResult();
}

function showCenterState(s) {

  document.getElementById('state-reading').classList.add('hidden');
  document.getElementById('state-locked').classList.add('hidden');
  document.getElementById('state-' + s).classList.remove('hidden');
}

  var el = document.getElementById(id);

}

function updateLockedDisplay() {
  var l = state.locked;
  document.getElementById('locked-side').textContent =
    l.side.toFixed(1) + '% ' + (l.sideDir === 'R' ? '→R' : '←L');
  document.getElementById('locked-ud').textContent =
    l.ud.toFixed(1) + '% ' + (l.udDir === 'flat' ? '—' : l.udDir === 'uphill' ? '⬆' : '⬇');

  var prevBadge = document.getElementById('prev-badge');
  if (state.previous) {
    var p = state.previous;
    document.getElementById('prev-side-val').textContent = p.side.toFixed(1) + '%';
    document.getElementById('prev-ud-val').textContent   = p.ud.toFixed(1) + '%' + (p.udDir === 'flat' ? '' : p.udDir === 'uphill' ? '⬆' : '⬇');
    prevBadge.classList.remove('hidden');
  } else {
    prevBadge.classList.add('hidden');
  }
}

// ═══════════════════════════════════════════════
// CALCULATION ENGINE
// ═══════════════════════════════════════════════
function calcPutt(lengthFeet, sidePct, sideDir, udDir, udPct, stimp) {
  var paces = lengthFeet / 3;
  var base  = (paces * 2) - 1;

  // Aim (inches) at given side slope %
  var aim = base * sidePct;

  // Up/downhill adjustment: 5% of aim per 1% slope
  var adj = aim * 0.05 * udPct;
  if (udDir === 'uphill')   aim -= adj;
  if (udDir === 'downhill') aim += adj;

  // Stimp adjustment: +10% per stimp above 10
  var stimpDelta = stimp - 10;
  aim = aim * (1 + 0.10 * stimpDelta);

  // Pace: multiplier depends on stimp AND direction
  // Stimp: 8=down14/up22, 9=down16/up20, 10=down18/up18, 11=down20/up16, 12=down22/up14
  var paceMultipliers = {
    8:  { downhill: 14, uphill: 22 },
    9:  { downhill: 16, uphill: 20 },
    10: { downhill: 18, uphill: 18 },
    11: { downhill: 20, uphill: 16 },
    12: { downhill: 22, uphill: 14 },
  };
  // For stimp values between table entries, interpolate
  var stimpLow  = Math.floor(stimp);
  var stimpHigh = Math.ceil(stimp);
  var fraction  = stimp - stimpLow;
  // Clamp to table range 8–12
  stimpLow  = Math.max(8, Math.min(12, stimpLow));
  stimpHigh = Math.max(8, Math.min(12, stimpHigh));
  var mLow  = paceMultipliers[stimpLow]  || paceMultipliers[10];
  var mHigh = paceMultipliers[stimpHigh] || paceMultipliers[10];
  var mDown = mLow.downhill + (mHigh.downhill - mLow.downhill) * fraction;
  var mUp   = mLow.uphill   + (mHigh.uphill   - mLow.uphill)   * fraction;

  var pace = 100;
  if (udDir === 'uphill')   pace = 100 + (mUp   * udPct);
  if (udDir === 'downhill') pace = 100 - (mDown * udPct);
  pace = Math.max(20, Math.min(220, pace));

  // Format aim distance
  var aimStr = '';
  if (aim < 0.5) {
    aimStr = 'Straight';
  } else if (aim < 12) {
    aimStr = aim.toFixed(1) + '"';
  } else {
    var ft  = Math.floor(aim / 12);
    var ins = Math.round(aim % 12);
    aimStr = ins === 0 ? ft + "'" : ft + "' " + ins + '"';
  }

  return { aim: aim, aimStr: aimStr, aimDir: sideDir, pace: Math.round(pace) };
}

function recalc() {
  var side, sideDir, ud, udDir;

  if (state.manualMode) {
    side    = state.manualSide;
    sideDir = state.manualSideDir;
    ud      = state.manualUD;
    udDir   = state.manualUDDir;
  } else if (state.locked) {
    side    = state.locked.side;
    sideDir = state.locked.sideDir;
    ud      = state.locked.ud;
    udDir   = state.locked.udDir;
  } else {
    clearResult();
    return;
  }

  var r = calcPutt(state.lengthFeet, side, sideDir, udDir, ud, state.stimp);

  // Aim
  if (r.aim < 0.5) {
    document.getElementById('aim-arrow').textContent = '—';
    document.getElementById('aim-arrow').style.color = 'var(--dim)';
  } else {
    document.getElementById('aim-arrow').textContent = sideDir === 'L' ? '←' : '→';
    document.getElementById('aim-arrow').style.color = 'var(--blue)';
  }
  document.getElementById('result-aim').textContent = r.aimStr;
  document.getElementById('result-aim-sub').textContent =
    r.aim < 0.5 ? 'No break' : (sideDir === 'L' ? 'Left' : 'Right') + ' of pin';

  // Pace color
  var col = r.pace >= 90 && r.pace <= 115 ? 'var(--green)'
          : r.pace < 60 || r.pace > 155    ? 'var(--red)'
          : 'var(--orange)';
  document.getElementById('result-pace').textContent = r.pace + '%';
  document.getElementById('result-pace').style.color = col;
  document.getElementById('result-pace-sub').textContent =
    r.pace < 80 ? 'Die at hole' : r.pace > 120 ? 'Firm pace' : 'Ideal pace';

  // Pace bar — scale: 0% at left, 160% at right edge
  var barPct = Math.min(100, (r.pace / 160) * 100);
  document.getElementById('pace-bar').style.width = barPct + '%';
  document.getElementById('pace-bar').style.background = col;
}

function clearResult() {
  document.getElementById('aim-arrow').textContent = '—';
  document.getElementById('aim-arrow').style.color = 'var(--dim)';
  document.getElementById('result-aim').textContent = '—';
  document.getElementById('result-aim-sub').textContent = 'Take a reading first';
  document.getElementById('result-pace').textContent = '—';
  document.getElementById('result-pace').style.color = 'var(--dim)';
  document.getElementById('result-pace-sub').textContent = '—';
  document.getElementById('pace-bar').style.width = '0%';
}
